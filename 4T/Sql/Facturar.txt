CREATE TABLE Clientes (
    cliDoc INT PRIMARY KEY,
    cliNombre VARCHAR(20),
    cliApellido VARCHAR(20),
	cliDireccion VARCHAR(20),
    cliTelefono VARCHAR(20)
);

CREATE TABLE Productos (
    proId INT PRIMARY KEY,
    proNombre VARCHAR(20),
    proValor FLOAT,
    proTipo VARCHAR(20)
);

CREATE TABLE Factura (
    facId INT PRIMARY KEY AUTO_INCREMENT,
    facFecha DATE,
    cliDoc INT,
    FOREIGN KEY (cliDoc) REFERENCES clientes (cliDoc)
);

CREATE TABLE DetalleFactura (
    detFacId INT PRIMARY KEY AUTO_INCREMENT,
    facId INT,
    proId INT,
    detFacCantidad INT,
    FOREIGN KEY (facId) REFERENCES factura (facId),
    FOREIGN KEY (proId) REFERENCES productos (proId)
);
